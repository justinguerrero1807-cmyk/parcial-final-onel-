! ================================================
! SW1 — VLANs + LACP Active + STP Root
! ================================================
enable
configure terminal
hostname SW1

! --- Crear VLANs ---
vlan 300
 name Estudiantes
 exit
vlan 500
 name Academicos
 exit
vlan 999
 name Nativa-Administrativa
 exit

! --- Puertos acceso VLAN 500 (Fa0/9 y Fa0/17) ---
interface range FastEthernet0/9, FastEthernet0/17
 switchport mode access
 switchport access vlan 500
 spanning-tree portfast
 no shutdown
 exit

! --- Trunk hacia R1 (Gig0/2 según imagen) ---
interface GigabitEthernet0/2
 switchport mode trunk
 switchport trunk native vlan 999
 switchport trunk allowed vlan 300,500,999
 no shutdown
 exit


! --- EtherChannel LACP hacia SW2 (SW1 = ACTIVE) ---
interface range FastEthernet0/23-24
 switchport mode trunk
 switchport trunk native vlan 999
 switchport trunk allowed vlan 300,500,999
 channel-protocol lacp
 channel-group 1 mode active
 no shutdown
 exit

interface Port-channel1
 switchport mode trunk
 switchport trunk native vlan 999
 switchport trunk allowed vlan 300,500,999
 no shutdown
 exit

! --- STP: SW1 es Puente Raíz para 300, 500, 999 ---
spanning-tree vlan 300 root primary
spanning-tree vlan 500 root primary
spanning-tree vlan 999 root primary

! --- IP de gestión (VLAN 999) ---
interface vlan 999
 ip address 192.168.99.10 255.255.255.0
 no shutdown
 exit
ip default-gateway 192.168.99.1

end
write memory

! ================================================
! BLOQUE SSH — pegar en R1, R2, R3, R4, SW1, SW2, SW3
! Cambia el hostname al principio según el equipo
! ================================================

enable
configure terminal

! --- Dominio (necesario para generar clave RSA) ---
ip domain-name campus.local

! --- Generar clave RSA 1024 bits ---
crypto key generate rsa general-keys modulus 1024
! (escribe "yes" si pregunta, luego Enter)

! --- SSH versión 2 ---
ip ssh version 2

! --- Usuarios (matricula sin privilegio, opelegrino admin) ---
username TU_MATRICULA privilege 1 secret Fe2024
username opelegrino privilege 15 secret Fe2024

! --- Enable secret (contraseña privilegiado) ---
enable secret Fe2024

! --- Consola local ---
line console 0
 password Fe2024
 login
 exit

! --- VTY solo SSH, autenticación local ---
line vty 0 4
 transport input ssh
 login local
 exit

! --- Cifrar contraseñas en texto plano ---
service password-encryption

end
write memory
